CREATE TABLE passport (id INT(11) AUTO_INCREMENT, passport_no VARCHAR(20), birth_date DATE, birth_place VARCHAR(10), birth_place_raw VARCHAR(20), 
                       authority VARCHAR(56), country VARCHAR(10),
                       issue_date DATE, issue_place VARCHAR(10), issue_place_raw VARCHAR(20), expiry_date DATE, 
                       line0 VARCHAR(56), line1 VARCHAR(56),
                       name VARCHAR(10), name_cn VARCHAR(10), name_cn_raw VARCHAR(20), sex VARCHAR(10), person_id VARCHAR(20),
                       type VARCHAR(10), src_country VARCHAR(10), PRIMARY KEY(id));

create table users (id INT(11) AUTO_INCREMENT, openid VARCHAR(30), PRIMARY KEY(id));

create table application (id int(11) AUTO_INCREMENT, applicant_id int(11), country varchar(10), visatype varchar(20), cur_status int(2), PRIMARY KEY (id))

create table applicant (id int(11) AUTO_INCREMENT, passport_id int(11), occupation varchar(56), otherinfo varchar(120), PRIMARY KEY (id))
